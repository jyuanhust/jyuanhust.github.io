# 189. 轮转数组

class Solution {
public:
	void rotate(vector<int>& nums, int k) {
		
        if(k % nums.size() == 0)
            return;
        else {
            k = k % nums.size();
        }
        
        int index = nums.size() - k;

		vector<int> res;

		for (int i = index; i < nums.size(); i++) {
			res.push_back(nums[i]);
		}

		for (int i = 0; i < index; i++) {
			res.push_back(nums[i]);
		}
		nums = res;
	}
};

  • 为什么使用置换方法的时候得需要循环，不能够逐个遍历逐个置换么？
